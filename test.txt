def extract_table_long(table):
    rows = []
    for cell in table.get("cells", []):
        rows.append({
            "row": cell.get("row_index"),
            "col": cell.get("column_index"),
            "text": cell.get("text"),
            "bbox": cell.get("bbox")
        })
    return rows

if tables_page4:

tables_page5 = [t for t in tables if t.get("page_number") == 5]

def normalize_cells(table):
    normalized = []
    for cell in table.get("cells", []):
        text = (cell.get("text") or "").strip()
        
        # heurystyka na scalone komórki
        if "\n" in text:
            parts = [p.strip() for p in text.split("\n") if p.strip()]
        else:
            parts = [text]
        
        for part in parts:
            normalized.append({
                "row": cell.get("row_index"),
                "col": cell.get("column_index"),
                "text": part,
                "bbox": cell.get("bbox")
            })
    return normalized

if tables_page5:
    table5_data = normalize_cells(tables_page5[0])
    print("PAGE 5 COMPLEX TABLE (NORMALIZED):")
    for r in table5_data:
        print(r)

7️⃣ Tekst pod zdjęciem

Zasada:

znajdź element image

sprawdź najbliższy tekst poniżej (po bbox y)

image_captions = []

for p in pages:
    elements = p.get("elements", [])
    images = [e for e in elements if e.get("type") == "image"]
    texts = [e for e in elements if e.get("type") == "text"]

    for img in images:
        img_bbox = img.get("bbox", [])
        if not img_bbox:
            continue
        
        img_bottom = img_bbox[1]

        possible_captions = [
            t for t in texts
            if t.get("bbox") and t.get("bbox")[3] < img_bottom
        ]

        if possible_captions:
            closest = sorted(
                possible_captions,
                key=lambda x: abs(x.get("bbox")[3] - img_bottom)
            )[0]

            image_captions.append({
                "page": p.get("page_number"),
                "caption": closest.get("content")
            })

print("IMAGE CAPTIONS:")
print(image_captions)
