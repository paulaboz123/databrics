1) Parse 1 PDF i wyciągnij elementy typu table z bbox
from pyspark.sql import functions as F

pdf_path = "dbfs:/FileStore/pdfs/moj.pdf"

pdf_df = (spark.read.format("binaryFile")
          .load(pdf_path)
          .select(F.col("path"), F.col("content").alias("pdf_bytes")))

# WYMUSZ v2.0 (ważne)
parsed_df = pdf_df.select(
    "path",
    F.expr("ai_parse_document(pdf_bytes, map('version','2.0'))").alias("parsed")
)

# Rozbij na elements[]
elements_df = parsed_df.select(
    "path",
    F.explode(F.col("parsed.document.elements")).alias("el")
)

tables_df = elements_df.where(F.col("el.type") == F.lit("table")).select(
    "path",
    F.col("el.id").alias("element_id"),
    F.col("el.content").alias("table_html"),
    F.col("el.bbox").alias("bbox")  # array: [{coord:[...], page_id:int}]
)

display(tables_df)

bbox w tym schemacie to tablica, gdzie każdy wpis ma coord (pikselowe współrzędne) i page_id (0-based).

2) Chunkowanie tabeli (HTML) po N wierszy + przypięcie bbox/page_id

To daje “chunki” tabeli (np. do RAG / dalszego parsowania), a do podświetlania używasz bbox całej tabeli (albo wielu bboxów, jeśli tabela jest pocięta na kilka elementów/stron).

import pandas as pd
from bs4 import BeautifulSoup

ROWS_PER_CHUNK = 20

def chunk_table_html(html: str, rows_per_chunk: int = 20):
    soup = BeautifulSoup(html or "", "html.parser")
    rows = soup.select("tr")
    chunks = []
    for i in range(0, len(rows), rows_per_chunk):
        chunk_rows = rows[i:i+rows_per_chunk]
        # zbuduj minimalny HTML dla chunka
        chunk_html = "<table>" + "".join(str(r) for r in chunk_rows) + "</table>"
        chunks.append({
            "row_start": i,
            "row_end": min(i + rows_per_chunk - 1, len(rows) - 1),
            "chunk_html": chunk_html
        })
    return chunks

# Zbierz do drivera tylko do testu na 1 PDF (dla wielu pdfów rób to w pandas_udf)
rows = tables_df.collect()

out = []
for r in rows:
    bbox = r["bbox"]  # lista dictów: {coord: [...], page_id: ...}
    for b in (bbox or []):
        page_id = b.get("page_id")
        coord = b.get("coord")  # zwykle [x0,y0,x1,y1] albo podobny format w px
        for ch in chunk_table_html(r["table_html"], ROWS_PER_CHUNK):
            out.append({
                "path": r["path"],
                "element_id": r["element_id"],
                "page_id": page_id,
                "coord": coord,
                "row_start": ch["row_start"],
                "row_end": ch["row_end"],
                "chunk_html": ch["chunk_html"]
            })

chunks_df = spark.createDataFrame(pd.DataFrame(out))
display(chunks_df)
